45、任务与软件定时器：软件定时器的创建和使用

本课程将详细介绍FreeRTOS中软件定时器的创建、配置和使用方法，以及软件定时器与任务之间的关系。

什么是软件定时器？

软件定时器是FreeRTOS提供的一种基于系统节拍的可配置定时机制，它允许开发者创建和管理多个定时器，每个定时器都可以在指定的时间间隔后执行回调函数。

与硬件定时器不同，软件定时器完全由FreeRTOS内核管理，不需要额外的硬件资源。它们特别适合处理非实时性要求较高的周期性任务。

// 软件定时器回调函数原型
void TimerCallback(TimerHandle_t xTimer);

软件定时器的类型

FreeRTOS支持两种类型的软件定时器：

  • 单次定时器：在指定的时间后执行一次回调函数，然后自动停止
  • 自动重载定时器：在指定的时间间隔后重复执行回调函数，直到被手动删除
定时器类型 特点 适用场景
单次定时器 执行一次后停止 延时操作、一次性事件
自动重载定时器 周期性重复执行 数据采样、状态检查、周期性任务

软件定时器的创建

创建软件定时器需要调用 xTimerCreate() 函数：

TimerHandle_t xTimerCreate(
    const char * const pcTimerName,     // 定时器名称（用于调试）
    const TickType_t xTimerPeriod,      // 定时周期（以节拍为单位）
    const UBaseType_t uxAutoReload,     // 自动重载标志
    void * const pvTimerID,             // 定时器ID
    TimerCallbackFunction_t pxCallbackFunction // 回调函数
);

创建示例：

// 创建单次定时器
TimerHandle_t xOneShotTimer = xTimerCreate(
    "OneShotTimer",     // 定时器名称
    pdMS_TO_TICKS(1000), // 1秒后触发
    pdFALSE,            // 单次定时器
    (void *)0,          // 定时器ID
    vOneShotTimerCallback // 回调函数
);

// 创建自动重载定时器
TimerHandle_t xAutoReloadTimer = xTimerCreate(
    "AutoReloadTimer",  // 定时器名称
    pdMS_TO_TICKS(500), // 每500ms触发一次
    pdTRUE,             // 自动重载
    (void *)1,          // 定时器ID
    vAutoReloadTimerCallback // 回调函数
);

软件定时器的使用

创建定时器后，需要启动它才能开始计时：

// 启动定时器（立即启动）
if (xTimerStart(xOneShotTimer, 0) != pdPASS) {
    // 启动失败处理
}

// 启动定时器（延迟启动）
if (xTimerStart(xAutoReloadTimer, pdMS_TO_TICKS(100)) != pdPASS) {
    // 启动失败处理
}

其他常用操作：

// 停止定时器
xTimerStop(xTimer, 0);

// 重置定时器
xTimerReset(xTimer, 0);

// 修改定时器周期
xTimerChangePeriod(xTimer, pdMS_TO_TICKS(2000), 0);

// 删除定时器
xTimerDelete(xTimer, 0);

软件定时器工作流程

软件定时器服务任务

FreeRTOS使用一个专门的定时器服务任务（daemon task）来管理所有软件定时器。这个任务在启动调度器时自动创建（如果配置了 configUSE_TIMERS 为1）。

定时器服务任务的主要职责：

  • 管理定时器命令队列
  • 处理定时器到期事件
  • 执行定时器回调函数
// 在FreeRTOSConfig.h中配置定时器服务任务
#define configUSE_TIMERS             1
#define configTIMER_TASK_PRIORITY    (configMAX_PRIORITIES - 1)
#define configTIMER_QUEUE_LENGTH     10
#define configTIMER_TASK_STACK_DEPTH (configMINIMAL_STACK_SIZE * 2)

软件定时器与任务的关系

软件定时器回调函数在定时器服务任务的上下文中执行，这意味着：

  • 回调函数不能调用可能导致阻塞的API（如 vTaskDelay()）
  • 回调函数应尽可能简短，避免影响其他定时器的执行
  • 如果需要执行耗时操作，应该在回调函数中通过队列、信号量等机制通知其他任务

软件定时器与任务的对比：

特性 软件定时器 任务
执行上下文 定时器服务任务 独立任务上下文
阻塞操作 不允许 允许
优先级 固定（服务任务优先级） 可配置
适用场景 周期性、轻量级操作 复杂、耗时操作

完整示例代码

下面是一个完整的软件定时器使用示例：

#include "FreeRTOS.h"
#include "task.h"
#include "timers.h"
#include "queue.h"

// 定义定时器ID
#define TIMER_ID_ONE_SHOT    0
#define TIMER_ID_AUTO_RELOAD 1

// 定时器句柄
TimerHandle_t xOneShotTimer, xAutoReloadTimer;

// 工作队列
QueueHandle_t xWorkQueue;

// 单次定时器回调函数
void vOneShotTimerCallback(TimerHandle_t xTimer)
{
    // 发送消息到工作队列
    uint32_t ulMessage = 100;
    xQueueSend(xWorkQueue, &ulMessage, 0);
    
    // 打印信息
    printf("单次定时器触发！\r\n");
}

// 自动重载定时器回调函数
void vAutoReloadTimerCallback(TimerHandle_t xTimer)
{
    // 发送消息到工作队列
    uint32_t ulMessage = 200;
    xQueueSend(xWorkQueue, &ulMessage, 0);
    
    // 打印信息
    printf("自动重载定时器触发！\r\n");
}

// 工作任务
void vWorkTask(void *pvParameters)
{
    uint32_t ulReceivedValue;
    
    for(;;)
    {
        // 等待工作队列消息
        if(xQueueReceive(xWorkQueue, &ulReceivedValue, portMAX_DELAY) == pdPASS)
        {
            // 处理接收到的消息
            printf("工作任务收到消息: %lu\r\n", ulReceivedValue);
            
            // 这里可以执行实际的工作
            // ...
        }
    }
}

// 主任务
void vMainTask(void *pvParameters)
{
    // 创建工作队列
    xWorkQueue = xQueueCreate(10, sizeof(uint32_t));
    
    // 创建单次定时器（1秒后触发）
    xOneShotTimer = xTimerCreate(
        "OneShotTimer",
        pdMS_TO_TICKS(1000),
        pdFALSE,
        (void *)TIMER_ID_ONE_SHOT,
        vOneShotTimerCallback
    );
    
    // 创建自动重载定时器（每2秒触发一次）
    xAutoReloadTimer = xTimerCreate(
        "AutoReloadTimer",
        pdMS_TO_TICKS(2000),
        pdTRUE,
        (void *)TIMER_ID_AUTO_RELOAD,
        vAutoReloadTimerCallback
    );
    
    // 启动定时器
    if(xOneShotTimer != NULL && xAutoReloadTimer != NULL)
    {
        xTimerStart(xOneShotTimer, 0);
        xTimerStart(xAutoReloadTimer, 0);
    }
    
    // 创建工作任务
    xTaskCreate(vWorkTask, "WorkTask", configMINIMAL_STACK_SIZE * 2, NULL, 2, NULL);
    
    // 主任务循环
    for(;;)
    {
        // 主任务可以执行其他操作
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

int main(void)
{
    // 创建主任务
    xTaskCreate(vMainTask, "MainTask", configMINIMAL_STACK_SIZE * 2, NULL, 1, NULL);
    
    // 启动调度器
    vTaskStartScheduler();
    
    // 不会执行到这里
    for(;;);
}
